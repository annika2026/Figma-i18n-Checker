performWidthValidation(textNodes, csvData) {
  // 将 CSV 转换为翻译对象
  const translations = {};
  csvData.forEach(row => {
    const key = row.key || row.Key || row['﻿key'];
    if (key) {
      translations[key] = {};
      Object.keys(row).forEach(column => {
        if (column.toLowerCase() !== 'key') {
          translations[key][column] = row[column];
        }
      });
    }
  });
  
  // 过滤出需要验证的节点
  const nodesToValidate = textNodes.filter(node => 
    translations[node.name]
  );
  
  // 调用 TextWidthChecker 进行精确验证
  const validationResults = this.textWidthChecker.validateI18nTexts(
    nodesToValidate, 
    translations
  );
  
  const report = this.textWidthChecker.generateReport(validationResults);
  
  return report;
}

calculateTextWidth(text, fontStyle) {
  // 参数验证（强制要求 fontSize, fontFamily, fontWeight）
  if (!fontStyle?.fontSize || !fontStyle?.fontFamily || 
      fontStyle?.fontWeight === undefined) {
    throw new Error('必须提供完整的字体信息');
  }
  
  const { fontSize, fontFamily, fontWeight, letterSpacing = 0 } = fontStyle;
  
  // 构建字体字符串
  const fontFamilyWithFallback = 
    this.fontFallbacks[fontFamily] || `${fontFamily}, sans-serif`;
  
  let fontString = '';
  if (fontWeight !== 400) fontString += `${fontWeight} `;
  fontString += `${fontSize}px ${fontFamilyWithFallback}`;
  
  // 设置 Canvas 字体
  this.ctx.font = fontString;
  
  // 测量文本
  const metrics = this.ctx.measureText(text);
  let width = metrics.width;
  
  // 应用字间距
  if (letterSpacing && text.length > 1) {
    width += letterSpacing * (text.length - 1);
  }
  
  console.log(`[精确测量] 文本: "${text}", 字体: ${fontString}, 宽度: ${width.toFixed(2)}px`);
  
  return width;
}

##### 5) **智能换行计算** (第295-394行)

**关键规则**: 只能在空格处换行

```javascript
calculateRequiredLines(text, containerWidth, fontStyle) {
  // 将文本拆分成单词（支持中英文混合）
  const words = this.splitTextIntoWords(text);
  
  let currentLineWidth = 0;
  let lines = 1;
  const spaceWidth = this.calculateTextWidth(' ', fontStyle);
  
  for (const word of words) {
    const wordWidth = this.calculateTextWidth(word, fontStyle);
    
    // 如果单个单词超过容器宽度，无法通过换行适配
    if (wordWidth > containerWidth) {
      console.log(`[换行检测] 发现无法换行的单词: "${word}"`);
      return Infinity; // 返回无穷大
    }
    
    // 检查是否需要换行
    const totalWidth = currentLineWidth + spaceWidth + wordWidth;
    
    if (totalWidth <= containerWidth) {
      currentLineWidth = totalWidth;
    } else {
      lines++;
      currentLineWidth = wordWidth;
    }
  }
  
  return lines;
}
```

**单词拆分** (第348-394行):
```javascript
splitTextIntoWords(text) {
  const words = [];
  let currentWord = '';
  
  for (let i = 0; i < text.length; i++) {
    const char = text[i];
    const charCode = char.charCodeAt(0);
    
    // 判断是否是中文字符（CJK）
    const isCJK = (charCode >= 0x4E00 && charCode <= 0x9FFF);
    
    if (isCJK) {
      // 中文字符单独作为一个"单词"
      if (currentWord) words.push(currentWord);
      words.push(char);
      currentWord = '';
    } else if (char === ' ' || char === '\n') {
      // 空格：结束当前单词
      if (currentWord) words.push(currentWord);
      currentWord = '';
    } else {
      // 英文字符：累积
      currentWord += char;
    }
  }
  
  if (currentWord) words.push(currentWord);
  return words;
}
```

##### 6) **批量验证 i18n 文本** (第407-566行)
```javascript
validateI18nTexts(textNodes, translations) {
  // 清空缓存
  this.clearCache();
  
  const results = {
    valid: [],
    overflow: [],
    summary: {
      total: 0,
      valid: 0,
      overflow: 0,
      languages: new Set()
    }
  };
  
  textNodes.forEach(node => {
    const nodeName = node.name;
    const containerWidth = node.absoluteBoundingBox?.width || 0;
    const maxLines = node.maxLines || 1;
    
    // 提取字体信息
    const fontStyle = {
      fontSize: node.style.fontSize,
      fontFamily: node.style.fontFamily,
      fontWeight: node.style.fontWeight,
      letterSpacing: node.style?.letterSpacing || 0
    };
    
    // 检查每种语言的翻译
    const nodeTranslations = translations[nodeName] || {};
    
    Object.entries(nodeTranslations).forEach(([language, text]) => {
      results.summary.languages.add(language);
      this.cacheStats.total++;
      
      // 缓存检查
      if (this.isInCache(nodeName, language, text, 
          fontStyle.fontFamily, fontStyle.fontSize, containerWidth)) {
        this.cacheStats.hits++;
        console.log(`  [${language}] "${text}" - ⚡ 已缓存，跳过检测`);
        return;
      }
      
      this.cacheStats.misses++;
      results.summary.total++;
      
      // 精确检查适配性
      const check = this.checkTextFit(text, containerWidth, fontStyle, maxLines);
      
      // 添加到缓存
      this.addToCache(nodeName, language, text, 
        fontStyle.fontFamily, fontStyle.fontSize, containerWidth);
      
      // 分类结果
      const result = { key: nodeName, language, text, check };
      
      if (check.fits) {
        results.valid.push(result);
        results.summary.valid++;
      } else {
        results.overflow.push(result);
        results.summary.overflow++;
      }
    });
  });
  
  results.summary.languages = Array.from(results.summary.languages);
  
  // 输出缓存统计
  const cacheStats = this.getCacheStats();
  console.log(`[验证完成] 总计: ${results.summary.total}, 适配: ${results.summary.valid}, 溢出: ${results.summary.overflow}`);
  console.log(`[缓存统计] 命中率: ${cacheStats.hitRate}`);
  
  results.cacheStats = cacheStats;
  return results;
}

```

## 🐛 已知问题

### 1. Canvas 架构问题
- **现象**: `ERR_DLOPEN_FAILED` - 架构不兼容
- **原因**: Canvas 模块为 x86_64，系统为 ARM64
- **解决**: `rm -rf node_modules && npm install`

### 2. 权限问题
- **现象**: `bad interpreter: Operation not permitted`
- **原因**: 沙盒权限限制
- **解决**: 使用 `node` 直接运行脚本

### 3. 精度差异
- **现象**: Canvas 测量与 Figma 渲染有 1-2px 差异
- **原因**: 不同渲染引擎、字体版本、亚像素渲染
- **解决**: 正常现象，已在报告中说明
